给定一个可能包含重复元素的整数数组 nums，返回该数组所有可能的子集（幂集）。

> 说明：解集不能包含重复的子集。
>
> 示例:
>
> 输入: [1,2,2]
> 输出:
> [
>   [2],
>   [1],
>   [1,2,2],
>   [2,2],
>   [1,2],
>   []
> ]



```javascript
var subsetsWithDup = function(nums) {
    let res = []
    let path = []
    let len = nums.length
    nums.sort((a,b)=>{return a-b})
    function backtrack(start, path){
        res.push(path.slice())
        for(let i=start; i<len; i++){
            //  去重
            if(i>start && nums[i] === nums[i-1])continue
            path.push(nums[i])
            backtrack(i+1, path)
            path.pop()
        }
    }
    backtrack(0, path)
    return res
};
```
