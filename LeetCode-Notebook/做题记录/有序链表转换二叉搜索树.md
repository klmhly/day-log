#### 有序链表转换二叉搜索树

```javascript
var sortedListToBST = function(head) {
    let res = []
    while(head){
        res.push(head.val)
        head = head.next
    }
    let len = res.length
    function helper(left, right){
        if(left > right){
            return null
        } 
        let mid = parseInt((left + right)/2)
        let root = {
            val: res[mid]
        }
        root.left = helper(left, mid-1)
        root.right = helper(mid+1, right)
        return root
    }
    return helper(0, len-1)
};
```
