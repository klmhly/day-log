77. 组合

```javascript
var combine = function(n, k) {
    let res = []
    function backtrack(first=1, cur=[]){
        if(cur.length === k){
            // 关键： cur.slice()   js 语言特性
            res.push(cur.slice())
        }
        for(let i=first; i<n+1; i++){
            cur.push(i)
            backtrack(i+1, cur)
            cur.pop()
        }
    }
    backtrack()
    return res
};
```
